
// album

curl --data "yearReleased=1980&genre=g3" http://localhost:3000/album/add

// get album that doesn't exist, should fail

// add track to album
curl --data "albumId=1&number=10&title=t10&duration=90&primaryArtist=a10" http://localhost:3000/album/addTrack

// delete album
curl -X PATCH --data "albumId=1&trackTitle=track1" http://localhost:3000/album/deleteTrack
